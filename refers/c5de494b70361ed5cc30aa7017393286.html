第二步：准备执行代码 

  1. import lombok.Getter;
  2. import org.apache.sshd.client.SshClient;
  3. import org.apache.sshd.client.channel.ClientChannel;
  4. import org.apache.sshd.client.channel.ClientChannelEvent;
  5. import org.apache.sshd.client.session.ClientSession;
  6. import org.apache.sshd.common.util.security.SecurityUtils;
  7.  
  8. import java.io.ByteArrayInputStream;
  9. import java.io.ByteArrayOutputStream;
  10. import java.io.IOException;
  11. import java.security.GeneralSecurityException;
  12. import java.util.ArrayList;
  13. import java.util.List;
  14.  
  15. public class SshdExecutor {
  16.  
  17.     SshClient client;
  18.     ClientSession session;
  19.  
  20.     @Getter
  21.     private String result;
  22.     @Getter
  23.     private String error;
  24.  
  25.     public SshdExecutor(String ip, Integer port, String user) {
  26.         client = SshClient.setUpDefaultClient();
  27.         client.start();
  28.         try {
  29.             session = client.connect(user, ip, port).verify(10 * 1000).getSession();
  30.         } catch (IOException e) {
  31.             e.printStackTrace();
  32.         }
  33.     }
  34.     
  35.     //密码方式
  36.     public SshdExecutor(String ip, Integer port, String user, String password) {
  37.         this(ip, port, user);
  38.         session.addPasswordIdentity(password);
  39.     }
  40.  
  41.     //公钥方式
  42.     public SshdExecutor(String ip, Integer port, String user, String keyName, String publicKey) {
  43.         this(ip, port, user);
  44.         try {
  45.             session.addPublicKeyIdentity(SecurityUtils.loadKeyPairIdentity(keyName, new ByteArrayInputStream(publicKey.getBytes()), null));
  46.         } catch (IOException e) {
  47.             e.printStackTrace();
  48.         } catch (GeneralSecurityException e) {
  49.             e.printStackTrace();
  50.         }
  51.     }
  52.     
  53.     //执行命令
  54.     public void execute(String command) {
  55.         try {
  56.             if (!session.auth().verify(10 * 1000).isSuccess()) {
  57.                 throw new Exception("auth faild");
  58.             }
  59.  
  60.             ClientChannel channel = session.createExecChannel(command);
  61.             ByteArrayOutputStream out = new ByteArrayOutputStream();
  62.             ByteArrayOutputStream err = new ByteArrayOutputStream();
  63.             channel.setOut(out);
  64.             channel.setErr(err);
  65.  
  66.             if (!channel.open().verify(10 * 1000).isOpened()) {
  67.                 throw new Exception("open faild");
  68.             }
  69.             List<ClientChannelEvent> list = new ArrayList<>();
  70.             list.add(ClientChannelEvent.CLOSED);
  71.             channel.waitFor(list, 10 * 1000);
  72.             channel.close();
  73.             result = out.toString();
  74.             error = err.toString();
  75.         } catch (Exception e) {
  76.             e.printStackTrace();
  77.         } finally {
  78.             if (client != null) {
  79.                 try {
  80.                     client.stop();
  81.                     client.close();
  82.                 } catch (IOException e) {
  83.                     e.printStackTrace();
  84.                 }
  85.             }
  86.         }
  87.     }
  88. }
