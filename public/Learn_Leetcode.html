014 Longest Common Prefix – Easy

    char* longestCommonPrefix(char** strs, int strsSize) 
    {
        if( strsSize <= 0 ) 
            return "";
        if( strsSize == 1)
            return strs[0];

        int  preLen = strlen(strs[0]);
        char* preStr = strs[0];
        int i,j;
        char c;

        for( i = 1; preLen > 0 && i < strsSize; ++i)
        {   char* pstr = strs[i];
            if( pstr == NULL ) 
                return "";
            j = 0;
            while( (j < preLen) && ((c=pstr[j]) == preStr[j]))
            {   ++j;
            }
            preLen = j;
        }

        preStr = malloc(preLen+1);
        memcpy(preStr, strs[0], preLen);
        preStr[preLen] = '\0';
        return preStr;
    }

    class Solution {
    public:
        string longestCommonPrefix(vector& strs) {

            int row = strs.size();
            if( row == 0 ) return "";
            if( row == 1)  return strs[0];

            int pos = 0;

                        match = false;
                }

                if( !match )
                    break;
            }

            return first.substr(0, pos);
        }

   After removing the second node from the end, the linked list becomes 1->2->3->5.

    Note:

    struct ListNode* removeNthFromEnd(struct ListNode* head, int n)
    {    
        if( head == NULL )
            return head;

        struct ListNode** array = malloc( (n+1)*sizeof(struct ListNode* ) );
        memset(array, 0, (n+1)*sizeof(struct ListNode*));

        int i = 0;
        struct ListNode* node = head;
        for( ;node != NULL; i %= n+1, node = node->next)
        {   array[i++] = node;
        } 

        struct ListNode* previous = array[i%(n+1)];
        struct ListNode* target = array[(i+1)%(n+1)];

        if( previous == NULL )
        {   head = target->next;
        }
        else
        {   previous->next = target->next;
        }

        free(target);
        return head;
    }
  • Solution–2016–05–02 C++: 04 ms – beats 20%
    • 
    class Solution {
    public:
        ListNode* removeNthFromEnd(ListNode* head, int n) {

            ListNode* node = head;
            while( n-- > 0 )
                node = node->next;

            if( node == NULL )
                return head->next;

            ListNode* previous = head;
            while( node->next != NULL)
            {
                node = node->next;
                previous = previous->next;
            }

            previous->next = previous->next->next;
            return head;
        }  
    };

020 Valid Parentheses – Easy

  • Description – LeetCode OJ

    • Given a string containing just the characters '(' , ')' , '{' , '}' , '[' and ']' , determine if the input string is valid.

    The brackets must close in the correct order, "()" and "()[]{}" are all valid but "(]" and "([)]" are not.

    bool isValid(char* s) 
    {
        if( s == NULL || *s == '\0' )
            return true;

        char stack[1024*64];
        int pos = 0;
        char c;

        while( (c=*s++) != '\0' )
        {
            char c2;
            switch(c)
            {
            case '(':
            case '[':
            case '{':
                stack[pos++] = c;
                continue; // continue while
            case ')': c2 = '('; break;
            case ']': c2 = '['; break;
            case '}': c2 = '{'; break;
            default:
                return false;
            }

            if( pos == 0 || stack[pos-1] != c2 )
                return false;
            --pos;
        }

        return pos == 0;
    }
  • Solution–2016–04–13 C++: 00 ms – beats 02%
    • 
    class Solution {
    public:

        bool isValid(string s) {

            stack ms;
            for( char c : s)
            {
                if( c == '(' || c == '{' || c == '[' )
                    ms.push(c);
                else if( ms.empty() || ms.top() != expect(c))
                    return false;
                else
                    ms.pop();
            }

            return ms.empty();
        }

        char expect( char c)
        {
            switch(c)
            {
              case ')': return '(';
              case '}': return '{';
              case ']': return '[';
              default : return 0;
            }
        }
    };

021 Merge Two Sorted Lists – Easy

  • Description – LeetCode OJ

    • Merge two sorted linked lists and return it as a new list. The new list should be made by splicing together the nodes of the first two lists.

  • Solution–2017–03–24 C: 6 ms – beats 31%

    /**
     * Definition for singly-linked list.
     * struct ListNode {
     *     int val;
     *     struct ListNode *next;
     * };
     */
    struct ListNode* mergeTwoLists(struct ListNode* l1, struct ListNode* l2) 
    {
        struct ListNode Temp;
        struct ListNode* l3 = &Temp;
        while( l1 != NULL && l2 != NULL )
        {
            if( l1->val > l2->val )
            {   l3->next = l2;
                l2 = l2->next;
            }
            else
            {   l3->next = l1;
                l1 = l1->next;
            }

            l3 = l3->next;
        }

        l3->next= (l1==NULL)? l2 : l1;
        return Temp.next;
    }
  • Solution–2016–04–02 C++: 08 ms – beats 68%
    • 
    class Solution {
    public:
        ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {

            ListNode tmp(0);
            ListNode* pnode = &tmp;

            while( l1 != NULL && l2 != NULL )
            {
                ListNode** pmin = (l1->val < l2->val) ? &l1 : &l2;

                pnode->next = *pmin;
                pnode = pnode->next;
                *pmin =(*pmin)->next;
            }

            if( l1 != NULL )
                pnode->next = l1;

            if( l2 != NULL )
                pnode->next = l2;

            return tmp.next;
        }
    };

024 Swap Nodes in Pairs – Easy

  • Description – LeetCode OJ

    • Given a linked list, swap every two adjacent nodes and return its head.

    For example,

    Given 1->2->3->4 , you should return the list as 2->1->4->3 Your algorithm should use only constant space. You may not modify the values in the list, only nodes itself can be changed.

  • Solution–2017–03–24 C: 3 ms – beats 6%
    • 
    /**
     * Definition for singly-linked list.
     * struct ListNode {
     *     int val;
     *     struct ListNode *next;
     * };
     */
    struct ListNode* swapPairs(struct ListNode* head) 
    {
        if( head == NULL || head->next == NULL )
        {   return head;
        }

        struct ListNode temp;
        struct ListNode* p1 = head;
        struct ListNode* p2 = NULL;
        struct ListNode* pp = &temp;

        while( p1 != NULL && p1->next != NULL )
        {
            p2 = p1->next;
            p1->next = p2->next;
            p2->next = p1;
            pp->next = p2;
            pp = p1;
            p1 = p1->next;
        }

        return temp.next;
    }
  • Solution–2016–04–24 C++: 04 ms – beats 01%
    • 
    class Solution {
    public:
        ListNode* swapPairs(ListNode* head) {

            if( head == NULL || head->next == NULL )
                return head;

            ListNode* next = head->next;
            ListNode* tail = next->next;
            next->next = head;
            head->next = swapPairs(tail);
            return next;
        }
    };

026 Remove Duplicates from Sorted Array – Easy

  • Description – LeetCode OJ

    • Given a sorted array, remove the duplicates in place such that each element appear only once and return the new length.

    Do not allocate extra space for another array, you must do this in place with constant memory.

    For example,

    Given input array nums = [1,1,2] ,

    Your function should return length = 2 , with the first two elements of nums being 1 and 2 respectively. It doesn't matter what you leave beyond the new length.

  • Solution–2017–03–24 C: 16 ms – beats 23%

    int removeDuplicates(int* nums, int numsSize) {

        if( nums == NULL || numsSize == 0 )
        {   return 0;
        }

        int pos = 0;
        while( pos < (numsSize-1) && nums[pos] != nums[pos+1])
            ++pos;

        int i = pos + 1;
        for(; i < numsSize; ++i )
        {   if( nums[pos] != nums[i] )
            {  nums[++pos] = nums[i];
            }
        }

        return pos + 1;
    }

036 Valid Sudoku – Easy

  • Description – LeetCode OJ

    • Determine if a Sudoku is valid, according to: Sudoku Puzzles – The Rules.

    The Sudoku board could be partially filled, where empty cells are filled with the character '.' .

    A partially filled sudoku which is valid.

    Note:

    A valid Sudoku board (partially filled) is not necessarily solvable. Only the filled cells need to be validated.

  • There are just 3 rules to Sudoku.

    • Each row must have the numbers 1–9 occuring just once.

    Each column must have the numbers 1–9 occuring just once.

    And the numbers 1–9 must occur just once in each of the 9 sub–boxes of the grid.

  • Solution–2017–03–24 C: 12 ms – beats 13%
    • 
bool isValidSudoku(char** board, int boardRowSize, int boardColSize) 
{
    if( boardRowSize !=9 ||  boardColSize != 9 )
    {   return false;
    }

    int count[3][9][9] = {0};
    int v;

    for( int r = 0; r < 9; ++r)
    {   for( int c = 0; c < 9; ++c)
        {   if( (v=board[r][c])=='.' )
            {   continue;
            }
            int vi = v - '1';
            if( vi < 0 
             || vi >= 9 
             || count[0][r][vi]++ != 0 
             || count[1][c][vi]++ != 0
             || count[2][r/3*3+c/3][vi]++ != 0 )
            {   return false;
            }
        }
    }
    return true;
}

        string countAndSay(int n) {
