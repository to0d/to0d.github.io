    bool isPalindrome(int x) 
    {
        if( x < 0 ) return false;

        int y = 0, z = x;
        for( ; z > 0 && y >= 0; z/=10 )
        {   y = y*10 + z%10;
        }

        return x == y;
    }

    class Solution {
    public:
        bool isPalindrome(int x) {

            if( x < 0 ) return false;

            int y = 0;
            int z = x;
            for(; z > 0; z/= 10 )
            {
                y = y * 10 + z % 10;
                if( y < 0 )
                    return false;
            }

            return x == y;
        }
    };

014 Longest Common Prefix – Easy

  • Description – LeetCode OJ

    • Write a function to find the longest common prefix string amongst an array of strings.

  • Solution–2017–03–24 C: 03 ms – beats 39%

    char* longestCommonPrefix(char** strs, int strsSize) 
    {
        if( strsSize <= 0 ) 
            return "";
        if( strsSize == 1)
            return strs[0];

        int  preLen = strlen(strs[0]);
        char* preStr = strs[0];
        int i,j;
        char c;

        for( i = 1; preLen > 0 && i < strsSize; ++i)
        {   char* pstr = strs[i];
            if( pstr == NULL ) 
                return "";
            j = 0;
            while( (j < preLen) && ((c=pstr[j]) == preStr[j]))
            {   ++j;
            }
            preLen = j;
        }

        preStr = malloc(preLen+1);
        memcpy(preStr, strs[0], preLen);
        preStr[preLen] = '\0';
        return preStr;
    }

    class Solution {
    public:
        string longestCommonPrefix(vector& strs) {

            int row = strs.size();
            if( row == 0 ) return "";
            if( row == 1)  return strs[0];

            int pos = 0;
            string& first = strs[0];

            for(; pos < first.size(); ++pos )
            {
                char c = first[pos];
                bool match = true;
                for( int r = 1 ; match && r < row ; ++ r)
                {
                    if( pos >= strs[r].size() || strs[r][pos] != c)
                        match = false;
                }

                if( !match )
                    break;
            }

            return first.substr(0, pos);
        }
    };

019 Remove Nth Node From End of List – Easy – Support <sup>th</sup> in MD @TODO

  • Question Description – LeetCode OJ

    • Given a linked list, remove the <sup>th</sup> n node from the end of list and return its head.

    For example, 

    
   Given linked list: 1->2->3->4->5, and n = 2.
   After removing the second node from the end, the linked list becomes 1->2->3->5.

    Note:

    Given n will always be valid.

    Try to do this in one pass.

  • Solution–2017–03–24 C: 06 ms – beats 19%
    • 
    struct ListNode* removeNthFromEnd(struct ListNode* head, int n)
    {    
        if( head == NULL )
            return head;

        struct ListNode** array = malloc( (n+1)*sizeof(struct ListNode* ) );
        memset(array, 0, (n+1)*sizeof(struct ListNode*));

        int i = 0;
        struct ListNode* node = head;
        for( ;node != NULL; i %= n+1, node = node->next)
        {   array[i++] = node;
        } 

        struct ListNode* previous = array[i%(n+1)];
        struct ListNode* target = array[(i+1)%(n+1)];

        if( previous == NULL )
        {   head = target->next;
        }
        else
        {   previous->next = target->next;
        }

        free(target);
        return head;
    }
  • Solution–2016–05–02 C++: 04 ms – beats 20%
    • 
    class Solution {
    public:
        ListNode* removeNthFromEnd(ListNode* head, int n) {

            ListNode* node = head;
            while( n-- > 0 )
                node = node->next;

            if( node == NULL )
                return head->next;

            ListNode* previous = head;
            while( node->next != NULL)
            {
                node = node->next;
                previous = previous->next;
            }

            previous->next = previous->next->next;
            return head;
        }  
    };

020 Valid Parentheses – Easy

  • Description – LeetCode OJ

    • Given a string containing just the characters '(' , ')' , '{' , '}' , '[' and ']' , determine if the input string is valid.

    The brackets must close in the correct order, "()" and "()[]{}" are all valid but "(]" and "([)]" are not.

  • Solution–2017–03–24 C: 3 ms – beats 23%
    • 
    bool isValid(char* s) 
    {
        if( s == NULL || *s == '\0' )
            return true;

        char stack[1024*64];
        int pos = 0;
        char c;

        while( (c=*s++) != '\0' )
        {
            char c2;
            switch(c)
            {
            case '(':
            case '[':
            case '{':
                stack[pos++] = c;
                continue; // continue while
            case ')': c2 = '('; break;
            case ']': c2 = '['; break;
            case '}': c2 = '{'; break;
            default:
                return false;
            }

            if( pos == 0 || stack[pos-1] != c2 )
                return false;
            --pos;
        }

        return pos == 0;
    }
  • Solution–2016–04–13 C++: 00 ms – beats 02%
    • 
    class Solution {
    public:

        bool isValid(string s) {

            stack ms;
            for( char c : s)
            {
                if( c == '(' || c == '{' || c == '[' )
                    ms.push(c);
                else if( ms.empty() || ms.top() != expect(c))
                    return false;
                else
                    ms.pop();
            }

            return ms.empty();
        }

        char expect( char c)
        {
            switch(c)
            {
              case ')': return '(';
              case '}': return '{';
              case ']': return '[';
              default : return 0;
            }
        }
    };
