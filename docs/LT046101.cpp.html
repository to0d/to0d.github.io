#include <lt_help/lt.h>

class Solution {
public:
    int hammingDistance(int x, int y) {
        return bitNumOf(x ^ y);
    }

    int bitNumOf(int n) {
        int sum = 0;
        for (; n!= 0; n &= (n-1))
            ++sum;
        return sum;
    }
};



void test(int x, int y)
{
    cout << "input: x=" << x << ", y=" << y;
    int r = Solution().hammingDistance(x, y);
    cout << "; output: " << r << ";" << endl;
}

int main(void)
{   test(1, 4);
}
