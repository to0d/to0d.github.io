#include <lt_help/lt.h>

class Solution {
public:
    vector<vector<int>> levelOrderBottom(TreeNode* root) { // refer LT010201.cpp
        vector<vector<int>> rst;

        vector<int> nodes;
        queue<TreeNode*> nq;
        nq.push(root);
        nq.push(NULL);
        while( !nq.empty() )
        {   TreeNode* node = nq.front(); nq.pop();
            if( node == NULL )
            {   if(nodes.empty())
                    break;
                rst.push_back(nodes);
                nodes.clear();
                nq.push(NULL);
            }
            else
            {   nodes.push_back(node->val);
                if( node->left != NULL )
                    nq.push(node->left );
                if( node->right != NULL )
                    nq.push(node->right );
            }
        }
        reverse(rst.begin(), rst.end());
        return rst;
    }
};
