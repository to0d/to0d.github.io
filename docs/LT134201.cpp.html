#include <lt_help/lt.h>



class Solution {
public:
    int numberOfSteps(int num) {
        int n = 0;
        while( num > 0 )
        {   if( num %2 == 1)
                --num;
            else
                num /= 2;
            ++n;
        }
        return n;
    }
};


void test(int num)
{
    cout << "input: " << num;
    int r = Solution().numberOfSteps(num);
    cout << "; output: " << r << ";" << endl;
}

int main(void)
{   test(14);
}
