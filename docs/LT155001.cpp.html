#include <lt_help/lt.h>

class Solution {
public:
    bool threeConsecutiveOdds(vector<int>& arr) {
        int n = 0;
        for( int v : arr )
            if( v%2 == 1 )
            {   if( ++n >= 3 )
                    return true;
            }
            else
                n = 0;
        return false;
    }
};



void test(vector<int> arr)
{
    cout << "input: ";
    outputVector(arr);
    bool r = Solution().threeConsecutiveOdds(arr);
    cout << "; output: " << r << ";" << endl;
}

int main(void)
{   test({1,2,34,3,4,5,7,23,12});
}
