Given a 32–bit signed integer, reverse digits of an integer.

Example 1: 

    Input: 123
    Output:  321

Example 2: 

    Input: -123
    Output: -321

Example 3: 

    Input: 120
    Output: 21

Note:

Assume we are dealing with an environment which could only hold integers within the 32–bit signed integer range. For the purpose of this problem, assume that your function returns 0 when the reversed integer overflows.

LT000701.cpp

#include <lt_help/lt.h>



// INT_MIN (-2147483648)
// INT_MAX (2147483647)

class Solution {
public:
    int reverse(int x) {
        if( x == INT_MIN )
            return 0;

        if( x < 0 )
            return -reverse(-x);

        long long r = 0;
        for(; x > 0; x /= 10)
        {   r = r*10+x%10;
            if( r > INT_MAX)
            return 0;
        }

        return (int)r;
    }
};

void test(int x)
{
    cout << "input: " << x;
    int r = Solution().reverse(x);
    cout << "; output: " << r << endl;
}

int main(void)
{
    test(123);
}
