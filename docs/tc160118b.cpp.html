};

class  List
{
    typedef Node<Ty>* 	node_ptr;
public:
    List():pHead(NULL),pTail(NULL),m_nCount(0){}

    int getCount() const { return m_nCount; }


private:
    List( const List& );
    List& operator = (const List& );

private:
    node_ptr pHead;
    node_ptr pTail;
    int m_nCount;
};

template<class Ty>
Node<Ty>* initList()
{
    Node<Ty>* pHead = new Node<Ty>();
    pHead->m_pNext = pHead;
    return pHead;
