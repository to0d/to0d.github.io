#include <lt_help/lt.h>

class Solution {
public:
#define isOdd(x) ((x)%2 == 1)
    int countOdds(int low, int high) {
        int count = 0;
        if( isOdd(high) )
        {   ++count;
            --high;
        }
        if( isOdd(low) )
        {   ++count;
            ++low;
        }
        return (high-low)/2 + count;
    }
};



void test(int low, int high)
{
    cout << "input: low=" << low << ", high=" << high;
    int r = Solution().countOdds(low, high);
    cout << "; output: " << r << ";" << endl;
}

int main(void)
{   test(3, 7);
}
