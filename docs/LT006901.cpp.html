#include <lt_help/lt.h>

class Solution {
public:
    int mySqrt(int x) {
        long low = 0, high = INT_MAX;
        while( low <= high )
        {   long m = (low+high)/2;
            long m2 = m*m;
            if( m2 == x )
                return m;
            if( m2 < x )
                low = m+1;
            else
                high = m-1;
        }
        return high;
    }
};



void test(int x)
{
    cout << "input: " << x;
    int r = Solution().mySqrt(x);
    cout << "; output: " << r << ";" << endl;
}

int main(void)
{   test(1024);
}
