example.flex

%{
    #include <stdio.h>
    #include <string.h>
    #include "y.tab.h"
%}

%%
[0-9]+          yylval.number=atoi(yytext); return NUMBER;
heater          return TOKHEATER;
heat            return TOKHEAT;
on|off          yylval.number = strcmp(yytext, "on"); return STATE;
target          return TOKTARGET;
temperature     return TOKTEMPERATURE;
[a-z0-9]+       yylval.string = strdup(yytext); return WORD;
\n              /* ignore end of line */
[ \t]+          /* ignore whitespace */
%%

example.yacc

%{
#include <stdio.h>
#include <string.h>

static char* heater;

    int yywrap()
    {
        return  1;
    }

    void yyerror()
    {
        printf(" parse error\n");
    }

    main()
    {
        yyparse();
    }
%}


%token  NUMBER TOKHEAT STATE TOKTARGET TOKTEMPERATURE TOKHEATER
%union
{
    int number;
    char *string;
}

%token <number> STATE
%token <number> NUMBER
%token <string> WORD

%%

commands
    : /* empty */
    | commands command
;

command
    : heat_switch
    | target_set
    | heater_select
;

heat_switch
    : TOKHEAT STATE
        {
            printf("\tHeat turned on or off\n");
        }
;

heater_select :
        TOKHEATER WORD
        {

        printf("\tSelected heater '%s'\n", $2);
        heater = $2;
        }
;

target_set:
    TOKTARGET TOKTEMPERATURE NUMBER
    {
    printf("\tHeater '%s' temperature set to %d\n", heater, $3);
    }
;

%%

makefile

CPPFLAGS=-I ../include/ctl -g -lfl
x.main.ctl.x: y.tab.o lex.yy.o
    gcc -o $@ $^ -lstdc++ -lfl	
    mv lex.yy.c lex.yy.c.tmp
    mv y.tab.c  y.tab.c.tmp
    mv y.tab.h  y.tab.h.tmp
    mv y.tab.o  y.tab.o.tmp
    mv lex.yy.o lex.yy.o.tmp

lex.yy.o:
    flex example.flex
    gcc -g -c lex.yy.c

y.tab.o:
    byacc -d  example.yacc
    gcc -g -c y.tab.c

clean:
    rm -f *.o
    rm -f *.tmp
    rm -f *.x
