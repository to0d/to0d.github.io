    Given an integer n, return true if it is a power of two. Otherwise, return false.

    An integer n is a power of two, if there exists an integer x such that n == 2ˆx.

    Example 1: 

    Input: n = 1
Output: true
Explanation: 20 = 1

LT023101.cpp

#include <lt_help/lt.h>

// 2  ==> 00000010
// 4  ==> 00000100
// 8  ==> 00001000
// 16 ==> 00010000
//
// 2^n ==> 0001000000000
//             |<- n ->|
class Solution {
public:
    bool isPowerOfTwo(int n) {
        return n > 0 && ( (n&(n-1)) == 0 );
    }
};



void test(int n)
{
    cout << "input: " << n << "; output: ";
    bool r = Solution().isPowerOfTwo(n);
    cout << r << endl;
}


int main(void)
{
    test(1);
    test(16);
    test(3);
}
