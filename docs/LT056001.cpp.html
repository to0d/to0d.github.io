#include <lt_help/lt.h>

class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int sum=0, ans=0;
        unordered_map<int,int> mp;
        mp[0] = 1;
        for(auto it:nums){
            sum += it;
            ans += mp[sum - k];
            mp[sum]++;
        }
        return ans;
    }
};



void test(vector<int> nums, int k)
{
    cout << "input: k=" << k << ", nums=";
    outputVector(nums);
    int r = Solution().subarraySum(nums, k);
    cout << "; output: " << r << ";" << endl;
}

int main(void)
{   test({1,2,3}, 3);
}
