#include <lt_help/lt.h>

class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) { // refer LT010201.cpp
        vector<vector<int>> rst;

        vector<int> nodes;
        queue<TreeNode*> nq;
        nq.push(root);
        nq.push(NULL);
        int level = 0;
        while( !nq.empty() )
        {   TreeNode* node = nq.front(); nq.pop();
            if( node == NULL )
            {   if(nodes.empty())
                    break;
                if( ++level %2 == 0 )
                    std::reverse(nodes.begin(), nodes.end());   // <== main change
                rst.push_back(nodes);
                nodes.clear();
                nq.push(NULL);
            }
            else
            {   nodes.push_back(node->val);
                if( node->left != NULL )
                    nq.push(node->left );
                if( node->right != NULL )
                    nq.push(node->right );
            }
        }
        return rst;
    }
};



void test(TreeNode* root)
{
    cout << "input: ";
    outputTree(root);
    vector<vector<int>> r = Solution().zigzagLevelOrder(root);
    cout << "; output: ";

}
