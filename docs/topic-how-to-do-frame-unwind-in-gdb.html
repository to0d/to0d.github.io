1. Test program

vi test.c

#include <stdio.h>

int func(int x)
{
    return x + 1;
}

void main()
{
    int y = 1, z;
    z = func( y );
    printf("hello, z=%d\n", z);
}

gcc -g test.c

./a.out
hello, z=2
