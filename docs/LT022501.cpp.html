#include <lt_help/lt.h>

class MyStack {
public:
    MyStack() {}

    void push(int x) {
        int size = mqueue.size();
        mqueue.push(x);
        while( size-- > 0 )
        {   mqueue.push(mqueue.front());
            mqueue.pop();
        }
    }

    int pop() {
        int val = mqueue.front();
        mqueue.pop();
        return val;
    }

    int top() {
        return mqueue.front();
    }

    bool empty() {
        return mqueue.empty();
    }

    queue<int> mqueue;
};


int main(void)
{   cout << "output: done!" << endl;
}
