    Given an integer num, return a string of its base 7 representation.

    Example 1: 

    Input: num = 100
Output: "202"

    Example 2: 

    Input: num = -7
Output: "-10"

    Constraints: 

    -107 <= num <= 107

LT050401.cpp

#include <lt_help/lt.h>

class Solution {
public:
    string convertToBase7(int num) {
        if( num == 0)
            return "0";

        static char buf[128];
        int pos = sizeof(buf);
        bool neg = false;
        if( num < 0 )
        {   neg = true;
            num = -num;
        }

        for(; num > 0; num /= 7)
            buf[--pos] = '0' + num%7;

        if(neg)
            buf[--pos] = '-';

        return string(buf+pos, sizeof(buf) - pos);
    }
};



void test(int num)
{
    cout << "input: " << num << "; output: ";
    string r = Solution().convertToBase7(num);
    cout << r << ";" << endl;
}

int main(void)
{
    test(0);
    test(100);
}
