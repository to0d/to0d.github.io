    Given n , how many structurally unique

    BST's (binary search trees) that store values 1... n ?

    For example,

    Given n = 3, there are a total of 5 unique BST's. 

    1         3     3      2      1
\       /     /      / \      \
3     2     1      1   3      2
/     /       \                 \
2     1         2                 3

LT009601.cpp

#include <lt_help/lt.h>

class Solution {
public:
    // f(n) = f(n-1)*f(0) + f(n-2)*f(1) ... + f(0)*f(n-1)
    int numTrees(int n) {
        vector<int> nums = {1, 1};
        int size = 2;
        while( size <= n ) {
            int num = 0;
            for(int i = 0;i < size; ++i )
                num += nums[i]*nums[size-i-1];
            nums.push_back(num);
            ++size;
        }
        return nums[n];
    }
};


void test(int n)
{
    cout << "input: " << n;
    int r = Solution().numTrees(n);
    cout << "; output: " << r << ";" << endl;
}


int main(void)
{
    test(3);
    test(1);
}
