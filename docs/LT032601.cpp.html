LT032601.cpp

#include <lt_help/lt.h>

class Solution {
public:
    bool isPowerOfThree(int n) {
        for(; n > 1; n = n/3)
            if( n%3 != 0 )
                return false;
        return n == 1;
    }
};



void test(int n)
{
    cout << "input: " << n << "; output: ";
    bool r = Solution().isPowerOfThree(n);
    cout << r << ";" << endl;
}


int main(void)
{
    test(27);
    test(0);
}
